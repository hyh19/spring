package com.smart.attr;

import java.util.HashMap;
import java.util.Map;

/**
 * 5.4.6 集合类型属性 Map
 */
public class Boss {
    private Map jobs = new HashMap();

    public Map getJobs() {
        return jobs;
    }

    public void setJobs(Map jobs) {
        this.jobs = jobs;
    }

    @Override
    public String toString() {
        return "Boss{" +
                "jobs=" + jobs +
                '}';
    }
}
